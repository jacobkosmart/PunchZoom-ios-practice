struct InfoPanelView: View {

var offset: CGSize

// MARK: -  CONTENT

  }

  Image(systemName: "arrow.up.and.down")
  Text("\(offset.height)")

  Spacer()
}
.font(.footnote)
.padding(8)
.background(.ultraThinMaterial)
.cornerRadius(8)
.frame(maxWidth: 420)
